#define _WINSOCK_DEPRECATED_NO_WARNINGS
#pragma warning(disable: 4996)

#include <iostream>
#include <winsock2.h>
#include <windows.h>
#include <cstring>

#pragma comment(lib, "ws2_32.lib")
using namespace std;

// Структура студента
struct Student {
    int id;
    char name[50];
    int grades[5];
    char status[100];
};

// Анализ оценок студента
void analyzeGrades(Student& s) {
    bool hasTwo = false;
    bool hasThree = false;

    for (int i = 0; i < 5; i++) {
        if (s.grades[i] == 2)
            hasTwo = true;
        else if (s.grades[i] == 3)
            hasThree = true;
    }

    if (hasTwo)
        strcpy(s.status, "Отчислен (есть оценка 2)");
    else if (hasThree)
        strcpy(s.status, "Обучается, но без стипендии");
    else
        strcpy(s.status, "Обучается. Назначена стипендия");
}

// Печать информации
void printStudent(const Student& s) {
    cout << "\nПолучен студент: " << s.name << "\nОценки: ";
    for (int g : s.grades)
        cout << g << " ";
    cout << "\nРешение деканата: " << s.status << endl;
}

int main() {
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);

    WSADATA wsa;
    SOCKET serverSocket;
    sockaddr_in serverAddr{}, clientAddr{};
    int clientSize = sizeof(clientAddr);

    if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0) {
        cout << "Ошибка WSAStartup\n";
        return 1;
    }

    serverSocket = socket(AF_INET, SOCK_DGRAM, 0);
    if (serverSocket == INVALID_SOCKET) {
        cout << "Ошибка создания UDP сокета\n";
        WSACleanup();
        return 1;
    }

    serverAddr.sin_family = AF_INET;
    serverAddr.sin_addr.s_addr = INADDR_ANY;
    serverAddr.sin_port = htons(9999);

    if (bind(serverSocket, (sockaddr*)&serverAddr, sizeof(serverAddr)) == SOCKET_ERROR) {
        cout << "Ошибка bind! Порт занят.\n";
        closesocket(serverSocket);
        WSACleanup();
        return 1;
    }

    cout << "UDP сервер деканата запущен (порт 9999)...\n";

    while (true) {
        Student student{};

        int received = recvfrom(
            serverSocket,
            (char*)&student,
            sizeof(student),
            0,
            (sockaddr*)&clientAddr,
            &clientSize
        );

        if (received == SOCKET_ERROR) {
            cout << "Ошибка recvfrom\n";
            continue;
        }

        analyzeGrades(student);
        printStudent(student);

        sendto(
            serverSocket,
            (char*)&student,
            sizeof(student),
            0,
            (sockaddr*)&clientAddr,
            clientSize
        );
    }

    closesocket(serverSocket);
    WSACleanup();
    return 0;
}
