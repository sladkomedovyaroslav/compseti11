#define _WINSOCK_DEPRECATED_NO_WARNINGS
#pragma warning(disable: 4996)

#include <iostream>
#include <winsock2.h>
#include <windows.h>
#include <string>
#include <cstring>

#pragma comment(lib, "ws2_32.lib")
using namespace std;

struct Student {
    int id;
    char name[50];
    int grades[5];
    char status[100];
};

void printResult(const Student& s) {
    cout << "\nОтвет деканата:\n";
    cout << "Студент: " << s.name << endl;
    cout << "Статус: " << s.status << endl;
    cout << "---------------------------\n";
}

int main() {
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);

    WSADATA wsa;
    SOCKET clientSocket;
    sockaddr_in serverAddr{};
    int serverSize = sizeof(serverAddr);

    if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0) {
        cout << "Ошибка WSAStartup\n";
        return 1;
    }

    clientSocket = socket(AF_INET, SOCK_DGRAM, 0);
    if (clientSocket == INVALID_SOCKET) {
        cout << "Ошибка создания UDP сокета\n";
        WSACleanup();
        return 1;
    }

    serverAddr.sin_family = AF_INET;
    serverAddr.sin_addr.s_addr = inet_addr("127.0.0.1");
    serverAddr.sin_port = htons(9999);

    cout << "UDP клиент студента запущен\n";

    int id = 1;
    string input;

    while (true) {
        Student student{};
        student.id = id++;

        cout << "\nВведите имя студента (exit для выхода): ";
        getline(cin, input);
        if (input == "exit") break;

        strcpy(student.name, input.c_str());

        cout << "Введите 5 оценок:\n";
        for (int i = 0; i < 5; i++) {
            cout << "Оценка " << i + 1 << ": ";
            cin >> student.grades[i];
        }
        cin.ignore();

        if (sendto(
            clientSocket,
            (char*)&student,
            sizeof(student),
            0,
            (sockaddr*)&serverAddr,
            serverSize
        ) == SOCKET_ERROR) {
            cout << "Ошибка отправки данных\n";
            continue;
        }

        if (recvfrom(
            clientSocket,
            (char*)&student,
            sizeof(student),
            0,
            (sockaddr*)&serverAddr,
            &serverSize
        ) == SOCKET_ERROR) {
            cout << "Сервер не ответил\n";
            continue;
        }

        printResult(student);
    }

    closesocket(clientSocket);
    WSACleanup();
    return 0;
}
